manifest.json 的结构

{
	manifest_version: 3,
	name: str,
	author?: str,
	description?: str,
	version: str,
	icons?: { "16": path, "32": path, "48": path, "128": path },

	// (1) 网站注入代码
	content_scripts?: {
		// 注入 css 的代码
		css?: css_path[],
		// 注入 js 的脚本（注：不能是 es6 语法，如 `export default` 的模块语法）
		js?: js_path[],
		// 需要注入的 js 或 css 代码的 origin 的模式串
		// 注：支持 * 通配符，如 '*://*.mozilla.org/*'
		matches: url[],
		// 排除的网站 origin 的模式串
		exclude_matches?: url[],
		// 注：支持 * 和 ? 通配符
		include_globs?: url[],
		exclude_globs?: url[],
		// 注入代码的时机
		run_at?:
			'document_start'	// DOM 正在加载
			| 'document_end'	// DOM 已完成 loading, 但还在加载资源
			| 'document_idle'	// (默认)DOM 及其资源加载完毕 
	}[],
	// (2) 弹出窗口配置
	action?: {
		default_icon?: path | { "16": path, "24": path, "32": path },
		default_title?: str,
		default_popup?: html_path
	},
	// (3) service worker => 事件处理程序
	background?: {
		service_worker: js_path,
	},
	// (4) 侧板，
	// 注：需要提供 'sidePanel' 权限
	side_panel?: {
		default_path: html_path
	},
	// (5) 选项页 => 用于管理扩展自定义选项
	// 注：建议使用 'storage' 权限
	options_page?: html_path,
	
	permissions?: enum[],	// 如：'storage', 'scripting', 'activeTab', 'sidePanel'
	host_permissions?: url[],
	optional_permissions?: enum[]
	optional_host_permissions?: url[],

	chrome_settings_overrides?: {
		homepage?: url,
		search_provider?: dict,
		startup_pages?: url[]
	},
	// 重写 chrome 的某些页面，注：一个插件只能提供以下某一个键值对
	chrome_url_overrides?: {
		// 重写 新标签页 chrome://newtab
		newtab?: path,
		// 重写 收藏夹 chrome://bookmarks
		bookmarks?: path,
		// 重写 edge://history
		history?: path
	},
	// 插件的主页
	homepage_url?: url
}

{
	content_scripts?: {
		// 注入 css 的代码
		css?: css_path[],
		// 注入 js 的脚本（注：不能是 es6 语法，如 `export default` 的模块语法）
		js?: js_path[],
		// 需要注入的 js 或 css 代码的 origin 的模式串，形如 <scheme>://<host>/<path> 或 file:///
		// scheme => http | https | * （* 只匹配 http 或 https）
		matches: url[],
		// 排除的网站 origin 的模式串
		exclude_matches?: url[],
		// 注：支持 * 和 ? 通配符
		include_globs?: url[],
		exclude_globs?: url[],
		// 注入代码的时机
		run_at?:
			'document_start'	// DOM 正在加载
			| 'document_end'	// DOM 已完成 loading, 但还在加载资源
			| 'document_idle',	// (默认)DOM 及其资源加载完毕 
		// 执行世界
		world?: 'ISOLATED' | 'MAIN',
		match_about_blank?: bool,
		match_origin_as_fallback?: bool,
		// 若为 false，则只有顶层 frame 会被注入代码；否则所有 frame 都会注入代码

相关定义：

  • sendMessage(message: any): Promise<any> => content script 发送消息给 service worker（短连接）
  • onMessage: Event<(message: any, sender: MessageSender, sendResponse: (response?: any) => void) => void> => service worker 接受 content script 的消息，并返回响应
  • runtime.connect(info?: ConnectInfo): Port => content script 发送消息给 service worker（长连接）
  • runtime.connect(extensionId: str, info?: ConnectInfo): Port
  • onConnect: Event<(port: Port) => void> => service worker 侦听 content script 的连接
  • connectNative(app: str): Port

Port 接口

{
	name: str,
	postMessage: (message: any)=>void,
	onMessage: PortMessageEvent,
	disconnect: ()=>void,
	onDisconnect: PortDisconnectEvent,
	sender?: MessageSender
}

例子 => 单次连接：

// (1) content_script.js
chrome.runtime.sendMessage({hello: 'world'})
    .then(res => console.log(res))

// 会报错？
chrome.tabs.query({active: true, lastFocusedWindow: true})
    .then(tabs => {
        chrome.tabs.sendMessage(tabs[0].id, {hello: 'world'})
            .then(res => console.log(res))
    })

// (2) service_worker.js
chrome.runtime.onMessage.addListener((message, sender, sendResponse)=>{
    console.log(sender.tab ? `from a content scripts: ${sender.tab.url}` : "from the extension")
    if(message.hello ===  'world'){
        sendResponse({hello: 'response'})
    }
})

onDisconnect 触发的时机

  • 调用过的 connect() 的 frame 被卸载
  • 接收端口的所有 frame 被卸载
  • 另一端调用了 disconnect()
