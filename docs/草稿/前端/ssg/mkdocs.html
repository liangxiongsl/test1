基本配置

	site_name: str,
	// 网站安装的位置，如：https://github.com/username/repo
	site_url?: str,
	// site_description 和 site_author 将添加到 html header 中
	site_description?: str,
	site_author?: str,
	copyright?: str
}

git 相关配置

{
	// 部署到 github pages 时选择的 remote，默认为 'origin'
	remote_name?: str,
	// 部署到 github pages 时选择的分支，默认为 'gh-pages'
	remote_branch?: str,
	// 若指定了，那么将在每个页面会有一个指向 repo_url 的链接
	repo_name?: 'GitHub' | 'Bitbucket' | 'GitLab',
	// 原则上需要放存储库的 url
	repo_url?: str,
	// '<repo_url>/<edit_uri>/foo/bar.md' 将指向编辑源文件 foo/bar.md 的 url
	// 对于 github 存储库：可以设置为 'edit/main/docs'
	// 注：有时 edit_uri 也可以时绝对 url
	edit_uri?: str,
	// 为 edit_uri 的扩展，但与其互斥
	// '{path}' 将转义为 foo/bar.md
	// '{path_noext}' 将转义为 foo/bar
	// `{path!q}` => 对链接进行 uri 编码（这样可能会降低 url 可读性）
	// 对于 github 存储库：可以设置为 'edit/main/docs/{path}'
	// 注：有时也可以直接与 obsidian 联动：'obsidian://open?vault=obsidian-vault&file={path_noext}'
	edit_uri_template?: str,
}

网站布局相关配置

{
	// 前端目录树对后端 .md 文件的目录树的映射配置
	// 若没有配置，则按照 'docs_dir' key 指定的目录对文件目录树进行导航布局
	nav?: Nodes,
	// 忽略 'docs_dir' key 指定的目录中匹配的文件
	// 例如：
	/*	exclude_docs: |  
		  *
		  !*.md
		  !*.css
		  node_modules/
		  !*.js
	*/
	// 注：匹配规则与 .gitignore 相同
	exclude_docs?: str,
	// 将文档标记为草稿：
	/*
		`mkdocs serve` 预览时不会在导航中出现这些文档，但可以通过链接访问
		`mkdocs serve --clean` 预览时不会在导航中出现这些文档，也不可通过链接访问
		`mkdocs build` 构建时不会对草稿文档进行处理
	*/
	draft_docs?: str,
	// 在导航（前端目录树）中将某些文档隐藏
	not_in_nav?: str,
	
	// 测试链接的相关配置
	// grade = 'warn' | 'info' | 'ignore'
	validation?: {
		nav?: {
			// `docs_dir` key 所指定的目录中的 .md 文档没有被 `nav` key 配置
			omitted_files?: 'info' | grade,
			// `nav` key 配置的资源 404 未找到或已被排除
			not_found?: 'warn' | grade,
			// `nav` key 配置出现 '/foo/bar.md' 等绝对路径
			// 注：若指定了 'relative_to_docs'，那么 '/' 会被解析为 `docs_dir` key 指定的目录
			absolute_links?: 'warn' | 'relative_to_docs' | grade
		},
		links?: {
			// .md 文档中 [](#anchor) 等链接的锚点无效
			anchors?: 'warn' | grade,
			unrecognized_links?: 'warn' | grade,
			not_found?: 'warn' | grade,
			absolute_links?: 'warn' | grade,
		},
		omitted_files?: 'warn' | grade,
		absolute_links?: 'warn' | grade,
		unrecognized_links?: 'warn' | grade,
		anchors?: 'warn' | grade
	}
}

注：exclude_docs, draft_docs, draft_docs 的匹配规则与 .gitignore 相同

Nodes 接口：

// 注：最多有 5 层
// 注2：若 .md 源文件没有一级标题，那么对应页面的标题为其文件名，否则为一级标题的内容
// 注3：map 的长度为 1
// 注4：str 可以是 path 对应 .md 文件对应的前端文件名，也可以是下一层的父目录
// 注5：前端目录树中文件的显示名的优先级：.md 文件名 < .md 文件内的一级标题 < .md 文件内元数据中 'title' key 指定的内容 < str 指定的 key
list<path | url | map<str, path | list>>

build 相关配置

{
	// 主题
	theme?: 'mkdocs' | 'readthedocs' | str | {
		// 主题名
		// 注：即使指定了 `custom_dir` key 也需要指定该 key
		name: str,
		// 自定义主题的目录，可以是相对于 '.yml' 父目录的目录，也可以是绝对目录
		custom_dir: path,
		// 如：'zh_CN'
		locate: enum,
		static_templates: list,
		// 特定于主题的配置
		<keys-specific-to-theme>: any
	},
	// markdown 源文件所在目录
	docs_dir?: 'docs' | path,
	// 构建的 web 文件输出的目录
	site_dir?: 'site' | str,
	extra_css?: list,
	extra_javascript?: list<str, str | {
		path: str,
		type?: enum,
		defer?: bool,
		async?: bool
	}>,
	extra_templates?: list,
	extra?: map<str, any>,
}

注：path 可以是相对于 .yml 父目录的目录，也可以是绝对目录

preview 相关配置

{
	// 指定 'mkdocs serve' 进行预览时，要侦听的其他目录
	// 注：默认侦听 `docs_dir` key 指定的目录下的文件
	watch?: list<str>,
	// 指定 http://example.com/foo/ 是否可以重定向为 http://example.com/foo.html
	use_directory_urls?: bool,
	// 指定在发出 warning 时是否定制进程
	strict?: bool,
	// 预览时使用的地址
	dev_addr?: '127.0.0.1:8000' | str
}

特殊 yaml tag

YAML Ain’t Markup Language (YAML™) revision 1.2.2

插入环境变量：

  • !ENV <env-val> => 会被替换为环境变量的值
  • !ENV [<env-val1>, <env-val2>, ..., <default-val>] => 匹配第一个存在的环境变量，否则返回 <default-val>

路径相关：

  • !relative => 当前 .md 文件的相对目录
  • !relative $docs_dir => docs_dir key 指定的目录
  • !relative $config_dir => 配置文件所在的目录
  • !relative $config_dir/foo/bar
