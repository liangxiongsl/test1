例子——下载图片

获取 dataURL（url 形式的数据）方式1——通过 fetch 和 FileReader

function runWithDataUrl(url: string, cb: (data_url: string)=>void){
    const fd = new FileReader()
    fd.onloadend = (e)=>{
        const data_url = e.target.result
        
        cb(data_url) // // data_url 形如 data:image/png;base64,iVBORw0KGgoAAAANSUhEUgAAAs ...
    }

    fetch(url, {responseType: 'blob'})
        .then(res => res.blob() )
        .then(blob => fd.readAsDataURL(blob) );
}

获取 dataURL 方式2——通过 canvas

function runWithDataUrl(url: string, cb: (data_url: string)=>void){
    const img = document.createElement('img')
    img.crossOrigin = 'anonymous' // => 允许跨域请求
    img.onload = ()=>{
        let canvas = document.createElement('canvas')
        canvas.width = img.width
        canvas.height = img.height
        let context = canvas.getContext('2d')
        context.drawImage(img, 0, 0, img.width, img.height)
        let data_url = canvas.toDataURL("image/png")

        cb(data_url)
    }
    img.src = url
}

获取 dataURL 方式3——需要提前声明标签 => <img src="..." id="img" crossorigin="anonymous">

function runWithDataUrl2(id, cb){
    const img = document.getElementById(id)
    let canvas = document.createElement('canvas')
    canvas.width = img.width
    canvas.height = img.height
    let context = canvas.getContext('2d')
    context.drawImage(img, 0, 0, img.width, img.height)
    let data_url = canvas.toDataURL("image/png")

    cb(data_url)
}

下载 dataURL 的内容到本地

function getFile(url){
    const a = document.createElement("a")
    a.href = url
    a.download = `${new Date().getTime()}`
    a.click()
}

总结 => runWithDataUrl(<url>, getFile) 可以直接下载文件到本地（文件名为当前时间戳）

IDBFactory 的方法：

  • open(name: str, version?: num): IDBOpenDBRequest

IDBOpenDBRequest < IDBRequest<IDBDatabase> 继承/自己的回调：

  • onsuccess: (this: IDBRequest<T>, ev: Event) => any => 数据库连接成功
  • onerror: (this: IDBRequest<T>, ev: Event) => any => 数据库连接失败
  • onupgradeneeded: (this: IDBOpenDBRequest, ev: IDBVersionChangeEvent) => any => 数据库创建或版本号更新时调用
  • onblocked: (this: IDBOpenDBRequest, ev: IDBVersionChangeEvent) => any

postMessage()

postMessage(message: any, targetOrigin: string, transfer?: Transferable[]): void
=> 可实现在浏览器上运行的 服务器 和 客户端 的交互

例子1

  1. 注册两个源的接受消息的回调（以 http://localhost:63342 和 http://localhost:5175 的通信为例）
// sender => http://localhost:63342
window.addEventListener('message',(e)=>{
	// 只接受特定源的消息
    if (e.origin !== 'http://localhost:5175') return  
	console.log(`i received: "[${e.origin}]${e.data}"`)
}, false)


// receiver => http://localhost:5175
window.addEventListener('message',(e)=>{
  if (e.origin !== 'http://localhost:63342') return
  console.log(`i received: "[${e.origin}]${e.data}"`)
  // 回复消息的例子（注意避免消息循环）
  e.source.postMessage("i received(reply)", e.origin)
  // 其他操作，比如：访问 localStorage 或 sessionStorage，然后返回数据
  // ...
}, false)
  1. http://localhost:63342 向发送 http://localhost:5175 消息
let iframe = open('http://localhost:5175')
// 等待网页打开完成，然后发送消息
setTimeout(()=>{
    iframe.postMessage('hello','http://localhost:5175')
}, 3000)

例子2 => 实现不同源网站之间的交互：（模拟 http 的 c/s 结构）

  • 客户端 / 服务器 的定义
// (1) 模拟客户端
window.addEventListener('message',(e)=>{
  // 用于处理请求和响应的函数
  let do_req = (type, data) => e.source.postMessage({type, data}, e.origin)
  let do_res = (data) => do_req('response', data)
  // 路由/中间件
  if (e.origin !== 'http://localhost:63342') return
  // 过滤不合法请求
  if (!e.data || !e.data.type || !e.data.data){
    return console.warn(`非法请求:${e.data}`)
  }

  // 处理请求
  let req = e.data
  switch (req.type){
    case 'response':
      console.log(req.data)
      break
  }
}, false)

// 开放请求接口
window.do_req = (url, type, data={})=>{
  let win = window.open(url)
  setTimeout(()=>win.postMessage({type: type, data: data}, url),1000)
}

// (2) 模拟服务器
window.addEventListener('message',(e)=>{
  // 用于处理请求和响应的函数
  let do_req = (type, data) => e.source.postMessage({type, data}, e.origin)
  let do_res = (data) => do_req('response', data)
  // 路由/中间件
  if (e.origin !== 'http://localhost:5175') return
  // 过滤不合法请求
  if (!e.data || !e.data.type || !e.data.data){
    return console.warn(`非法请求:${e.data}`)
  }

  // 处理请求
  let req = e.data
  switch (req.type){
    case 'get':
      do_res(JSON.parse(localStorage.getItem('data')))
      close()
      break
    case 'post':
      localStorage.setItem('data', JSON.stringify(req.data))
      do_res(`saved => 'data: ${JSON.stringify(req.data)}'`)
      close()
      break
    case 'response':
      console.log(req.data)
      break
  }
}, false)
